#version 330 core

uniform vec4 ParticleColor;

in vec3 PosCamera;
in vec3 PosWorld;

out vec4 FragColor;

void main()
{
	vec3 n;
	n.xy = gl_PointCoord * vec2(2.0, -2.0) + vec2(-1.0, 1.0);
	float mag = dot(n.xy, n.xy);
	if (mag > 1.0) discard;
	n.z = sqrt(1.0 - mag);

	vec3 lightDir = normalize(vec3(10.0f, 10.0f, 10.0f) - PosWorld);

    float diffuse = max(0.0, dot(lightDir, n));

    vec4 tempColor = vec4(0.6 * diffuse * vec3(ParticleColor) + vec3(0.05, 0.05, 0.2f), ParticleColor.a);
    FragColor = tempColor;
}
